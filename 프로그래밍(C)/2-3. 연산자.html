Chapter 2. C 프로그래밍의 기초

3. 연산자 (Operator)

데이터를 가공하고 계산하기 위한 연산자의 종류를 배우고, 우선순위와 결합 법칙을 통해 복합적인 수식을 이해합니다.

Think & Objective

"컴퓨터는 연산의 순서를 어떻게 결정할까요?"

  • 연산, 비교, 논리 등의 여러 종류의 연산자의 의미를 알고 프로그래밍에 적절하게 사용할 수 있다.
  • 다양한 연산자를 활용하여 프로그래밍 할 수 있다.

1. 산술 및 대입 연산자

86.c 연산자 우선순위

#include <stdio.h>
int main()

#include <stdio.h>

2. 증감 연산자 (++, --)

88.c 전치 vs 후치 연산의 차이
88.c 
#include <stdio.h>
int main()
{
	int i = 13, j = 5, temp,
		a = 10, b = 20, c = 30, d = 40;
	temp = ++i; printf("%d %d\n", temp, i);
	temp = i++; printf("%d %d\n", temp, i);
	temp = --j; printf("%d %d\n", temp, j);
	temp = j--; printf("%d %d\n", temp, j);
	++a; b++; --c; d--;
	printf("%d %d %d %d", a, b, c, d);
	return 0;
}

90-1.c 논리 연산자
90-1.c 
#include <stdio.h>
int main()
{
	int i = 13, j = 5, temp;
	printf("%d %d\n", i == 13 && j == 5, i == 13 && j == 4);
	printf("%d %d\n", i == 13 || j == 4, i == 12 || j == 4);
	temp = i; printf("%d ", !temp);
	temp = 0; printf("%d", !temp);
	return 0;
}
90-2.c 주의: 단락 회로 평가(Short-Circuit)
90-2.c 
#include <stdio.h>
int main()
{
	int i = 13, j = 5;
	if (i < 10 && ++j > 3) {}
	printf("%d %d\n", i, j);
	return 0;
}

4. 비트 연산자

91-1.c 비트 논리 연산자 (&, |, ^, ~)
91-1.c 
#include <stdio.h>
int main()
{
	int i = 13;  // 00000000 00000000 00000000 00001101
	int j = 5;   // 00000000 00000000 00000000 00000101
	printf("%d %d %d %d", i&j, i | j, i^j, ~i);
	return 0;
}
91-2.c 시프트 연산자 (<<, >>)
91-2.c 
#include <stdio.h>
int main()
{
	int i = 13;  // 00000000 00000000 00000000 00001101
	int j = 5;   // 00000000 00000000 00000000 00000101
	printf("%d %d %d %d", i << 2, i >> 2, j << 3, j >> 3);
	return 0;
}

5. 연산자 응용 예제 종합

92.c 정기예금 이자 계산기
92.c 
#include <stdio.h>
int main()
{
	int money, duration;
	double rate;
	double result;
	printf("정기예금 금액을 입력해 주세요 :");
	scanf("%d", &money);
	printf("정기예금 이율을 입력해 주세요 :");
	scanf("%lf", &rate);
	printf("정기예금 기간을 입력해 주세요 :");
	scanf("%d", &duration);
	result = money + money * duration * 0.01 * rate;
	printf("%d원을 %.1f%%이율로 %d년간 예치 후 원리 합계는 %.0f입니다.",
		money, rate, duration, result);
	return 0;
}
93.c 사칙연산 계산
93.c 
#include <stdio.h>
int main()
{
	int a, b;
	printf("두 수를 입력하세요 :");
	scanf("%d %d", &a, &b);
	printf("%d + %d = %d\n", a, b, a + b);
	printf("%d - %d = %d\n", a, b, a - b);
	printf("%d * %d = %d\n", a, b, a * b);
	printf("%d / %d = %f\n", a, b, (float)(a / b));
	printf("%d %% %d = %d" , a, b, a % b);
	return 0;
}
94-1.c 구속 변환 (마일 → km)
94-1.c 
#include <stdio.h>
int main()
{
	int mileSpeed;
	double kmSpeed;

	printf("%d마일은 %.3fkm입니다.", mileSpeed, kmSpeed);
	return 0;
}

95-1.c 더치페이 계산 1 (균등 분배)
95-1.c 
#include <stdio.h>
int main()
{
	int price, person;
	printf("총 금액 : ");
	scanf("%d", &price);
	printf("총 인원 : ");
	scanf("%d", &person);
	printf("%d명이 각각 %d원씩 지불", person, price / person);
	return 0;
}
95-2.c 더치페이 계산 2 (나머지 몰아주기)
95-2.c 
#include <stdio.h>
int main()
{
	int price, person;
	printf("총 금액 : ");
	scanf("%d", &price);
	printf("총 인원 : ");
	scanf("%d", &person);
	printf("%d명이 각각 %d원씩 지불", person-1, price / person);
	printf(", 1명 %d원 지불", price - (price/person)*(person-1));
	return 0;
}

핵심 요약

산술(%, /, *)
증감(++i vs i++)
논리(&&, ||, !)
비트(&, |, ^, ~)
